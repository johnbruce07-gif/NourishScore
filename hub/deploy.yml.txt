name: Deploy NourishScore to GitHub Pages

on:
  push:
    branches: [ main ]
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: "pages"
  cancel-in-progress: false

jobs:
  validate:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Validate all HTML files exist
        run: |
          echo "Checking required files..."
          test -f index.html              && echo "✅ index.html"
          test -f kitchen/index.html      && echo "✅ kitchen/index.html"
          test -f kitchen/manifest.json   && echo "✅ kitchen/manifest.json"
          test -f kitchen/sw.js           && echo "✅ kitchen/sw.js"
          test -f coach/index.html        && echo "✅ coach/index.html"
          test -f coach/manifest.json     && echo "✅ coach/manifest.json"
          test -f coach/sw.js             && echo "✅ coach/sw.js"
          test -f hub/index.html          && echo "✅ hub/index.html"
          test -f hub/admin.html          && echo "✅ hub/admin.html"
          test -f hub/manifest.json       && echo "✅ hub/manifest.json"
          test -f hub/sw.js               && echo "✅ hub/sw.js"

      - name: Validate manifest JSON files
        run: |
          python3 -c "import json; json.load(open('kitchen/manifest.json'))" && echo "✅ kitchen/manifest.json valid JSON"
          python3 -c "import json; json.load(open('coach/manifest.json'))"   && echo "✅ coach/manifest.json valid JSON"
          python3 -c "import json; json.load(open('hub/manifest.json'))"     && echo "✅ hub/manifest.json valid JSON"

      - name: Check Supabase is configured in hub
        run: |
          if grep -q "YOUR_SUPABASE_URL" hub/index.html; then
            echo "⚠️  Warning: hub/index.html still contains placeholder Supabase credentials"
            echo "   Update SUPABASE_URL and SUPABASE_ANON_KEY before deploying NourishHub"
          else
            echo "✅ Supabase credentials configured"
          fi

  deploy:
    needs: validate
    runs-on: ubuntu-latest
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Pages
        uses: actions/configure-pages@v4

      - name: Upload artifact
        uses: actions/upload-pages-artifact@v3
        with:
          path: '.'

      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4
